Question

Create a custom filter function that mimics the behavior of Array.prototype.filter(). The function should take an array and a callback function as parameters, and return a new array containing only the elements for which the callback returns true.

Your Answer
function customFilter(array, callback) {
    const result = [];
    for (let i = 0; i < array.length; i++) {
        if (callback(array[i], i, array)) {
            result.push(array[i]);
        }
    }
    return result;
}

// Example usage:
const numbers = [1, 2, 3, 4, 5];
const evenNumbers = customFilter(numbers, x => x % 2 === 0);
console.log(evenNumbers); // [2, 4]
